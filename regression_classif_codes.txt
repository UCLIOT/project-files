clcclearclose all load sdata3.txt data_x = sdata3(:,1:3);data_y = sdata3(:,end);X = [ones(50,1) data_x];Y = data_y;% regression[b,bint,r,rint,stats] = regress(Y,X,0.05)rcoplot(r,rint) TY      =   zeros(length(data_y),5); for i = 1:length(data_y)    TY(i,data_y(i)) = 1;end data_y = TY;  sep     =   1500;train_x = data_x(1:sep,:);test_x = data_x(sep+1:end,:);train_y = data_y(1:sep,:);test_y = data_x(sep+1,:);  % normalize[train_x, mu, sigma] = zscore(train_x); %% plotting functionalityrand('state',0)nn = nnsetup([3 100 5]);opts.numepochs         = 50;          %  Number of full sweeps through datann.output              = 'softmax';    %  use softmax outputopts.batchsize         = 1;            %  Take a mean gradient step over this many samplesopts.plot              = 1;            %  enable plotting nn = nntrain(nn, train_x, train_y, opts); [er, bad, labels] = nntest(nn, test_x, test_y);
